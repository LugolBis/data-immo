Data Format from the DVF API :

id | idmutation | idpar ? Primary key
vefa (Boolean)
datemut (String) -> Date
typologie dict(
    "libelle" (String)
)

dispositions list[
    dict(
        parcelles list[
            dict(
                dcnt list[
                    dict(
                        classes [
                            dict(
                                classes (String) tq "01", "02", etc. C'est un simple compteur qui s'incrémente et qui est partagé entre les dict de la liste
                                surface (Number)
                                libclasse (String) tq "terres", "vignes", "bois", "jardins"
                            )
                        ]
                        surface (Number) # C'est la somme des surfaces de chaque classe !
                        regroupement (String) "agri", "nat", etc. C'est l'abréviation du regroupement des classes
                        libregroupement (String) "agricole", "naturel", etc. C'est le nom du regroupement des classes
                    )
                ]

                lots ?? list[
                    dict(
                        nolot ??? (String) tq "17", "19", etc.
                        "scarrez" (???)
                        iddispoloc (Number)
                        iddispolot (Number)
                    )
                ]
                sufs list[
                    dict(
                        sterr ?? (Number)
                        nbsufidt ?? (Number)
                        nature_culture dict(
                            natcult (String)
                            libcult (String)
                        )
                        nature_culture_speciale list[???]
                    )
                ]

                idpar (String)
                locaux list[
                    dict(
                        idloc ?? (String)
                        sbati ?? (Number)
                        adresse dict(
                            ..... # Same as below
                        )
                        indentloc (???)
                        nbpprinc (Number)
                        datemutpre (???)
                        datemutsui (???)
                        iddispoloc (Number)
                        l_idmutpre list[???]
                        l_idmutsui list[???]
                        type_local dict(
                            codetyploc ?? (Number)
                            libtyploc ?? (String)
                        )
                    )
                ]
                volumes list[???]
                adresses list[
                    dict(
                        btq (???)
                        voie (String)
                        novoie (Number)
                        codvoie (String)
                        commune (String)
                        typvoie (???)
                        codepostal (String)
                        idadrinvar (String) # Fusionne toutes les infor de l'adresse, et met des dollars entre les variables
                        # Il serait idéal de faire un mini parseur pour remplacer les '$' par des ','
                    )
                ]
                parcvendue (Boolean)
            )
        ]
        valeurfonc (Number)
    )
]

nature_mutation dict(
    "code" (Number)
    "libelle" (String)
